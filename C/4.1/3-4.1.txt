   /*Escreva um programa que preencha dois vetores A e B (cada um com tamanho 50) com valores

int vetor1[50];/*Primeiro vetor*/
int vetor2[50];/*Segundo vetor*/

  printf("Digite o %dº valor do vetor 2:", k+1);

}
